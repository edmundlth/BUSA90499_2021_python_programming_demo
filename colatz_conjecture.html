Background: https://en.wikipedia.org/wiki/Collatz_conjecture $$ f(n) = \frac{n}{2} \quad\text{ if $n$ is divisible by 2}\\ f(n) = 3n + 1 \quad \text{ otherwise} $$

%%time

# This "memo-isation" is a programming technique call dynamic programming used to 
# speed up computation with a lot of recursive structure
memo = {1 : 0}

def iterate(n):
    if n % 2 == 0:
        return n // 2
    else:
        return (3 * n + 1)

N = 50000
remaining = set(range(1, N)) - set(memo.keys())
num = 10 if not remaining else min(remaining)
chain = [num]
c = num
while c < N and remaining:
    result = iterate(num)
    chain.append(result)
    if result in memo:
        for i in range(len(chain) - 2, -1 , -1):
            memo[chain[i]] = memo[chain[i + 1]] + 1
        remaining = remaining - set(memo.keys())
        if not remaining:
            break
        num = min(remaining)
        c = num
        chain = [num]
    else: 
        num = result


print(len(memo))
